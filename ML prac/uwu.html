#making some plotting functions
def plotpoints(X, Y):
    data = np.concatenate((X, Y), axis = 1)
    for rec in data:
        if rec[3]<=-1: plt.plot(rec[1], rec[2], 'rX', ms = 8) #'r' is red, 'x' is x shaped dot
        else: plt.plot(rec[1],rec[2], 'go', ms = 8) #'g' is green, 'o' is circle shaped dot
            
def plotline(w, fromwhere, towhere):
    x1 = np.arange(fromwhere, towhere)
    x2 = -1/w[2]*(w[1]*x1 + w[0])
    plt.plot(x1, x2, linewidth = 4)

#squisher function
def squish(Z):
    Y = []
    for ele in Z:
        if ele > 0: Y.append([1])
        else: Y.append([-1])
    return np.array(Y)

class Perceptron():
    def __init__(self, X):
        np.random.seed(7)
        self.w = np.random.randint(-3,5, (len(X[0]), 1))
        print('new weights:\n', self.w)
    
    def train(self, X, Y, alpha = 0.06):
        for i in range(len(X)):
            self.w = self.w + (alpha*Y[i]*X[i]).reshape(len(X[i]), 1)
        #print("Updated weights :\n", self.w)               #uncomment this to see weight change every epoch
        
    def predict(self, X):
        z = np.matmul(X, self.w)
        #print('\nz = ', z)
        y = squish(z)
        return y
    
    def plot(self, X, Y):
        plotpoints(X, Y)
        plotline(self.w, min(X[:,1])-1, max(X[:, 1])+2)
